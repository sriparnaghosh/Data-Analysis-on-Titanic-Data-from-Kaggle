def intial_survived(column): # defing the function to get the statistical data
    #getting the total number of saved indiviuals grouped by the passing column who survived
    survival_column = titanic_data.groupby(column)['Survived'].sum()
    # to get the total number of people wo were intially onboard
    total_column = titanic_data.groupby(column).size()
    # to concat the two data
    statistic_data = pd.concat([total_column,survival_column],axis=1)
    # to rename the column of the concatted dataset
    statistic_data = statistic_data.rename(columns = {0:'Intially'})
    #to find the portion of people saved
    statistic_data['portion_saved'] = survival_column / total_column
    # to find the probability of people saved
    statistic_data['probability_saved'] = statistic_data.portion_saved / statistic_data['portion_saved'].sum() 
    
    return statistic_data

# Ploting the  function for a pie plot 
def pie_plot(values,explode,labels,colors):
    plt.pie(values, explode=explode, labels=labels, colors=colors,
            autopct='%1.1f%%', shadow=True, startangle=180)

    plt.axis('equal')
    plt.show()

Limitations¶

There were about more than 2200 people on the board and the sample has only 891 passengers so the dataset may contain a particular data more. The dataset didn't have any crew member details, which I felt was the most important as they had the least chances of survival keeping the service ethics into consideration.

Here we see that the the portion saved is more between 0-10 and 50-60 and this is what we expected as children and seior citizens are to be saved first. The frequency is more between 30-60 because there are more people of this age group
