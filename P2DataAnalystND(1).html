4)How many females were there versus males on the ship?

#Generic function for seperating out grouped data for onborad and survived passengers and ther Survival rates

def grouping_statistic(df,column):
    
    #adding up survived per group
    survival_data = df.groupby(column)['Survived'].sum()
    
    #calculating onboard per group
    onboard_data = df.groupby(column).size()
    
    #merging the two data
    statistic = pd.concat([survival_data,onboard_data],axis=1)
    statistic = statistic.rename(columns = {0:'Onboard'})    #renaming the column
    
    #Calculating survival rate for each category
    statistic['Survival_rate'] = statistic.Survived / statistic.Onboard
    
    return statistic

# Plot function for a pie plot
def pieplot(values,explode,labels,colors):
    plt.pie(values, explode=explode, labels=labels, colors=colors,
            autopct='%1.1f%%', shadow=True, startangle=100)

    plt.axis('equal')
    plt.show()
